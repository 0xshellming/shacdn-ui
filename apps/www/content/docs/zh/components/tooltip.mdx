## 用法

```tsx
import {
  Tooltip,
  TooltipContent,
  TooltipProvider,
  TooltipTrigger,
} from "@/components/ui/tooltip"
```

```tsx
<TooltipProvider>
  <Tooltip>
    <TooltipTrigger>悬停</TooltipTrigger>
    <TooltipContent>
      <p>添加到库中</p>
    </TooltipContent>
  </Tooltip>
</TooltipProvider>
```
