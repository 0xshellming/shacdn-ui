## 用法

```tsx
import {
  Menubar,
  MenubarContent,
  MenubarItem,
  MenubarMenu,
  MenubarSeparator,
  MenubarShortcut,
  MenubarTrigger,
} from "@/components/ui/menubar"
```

```tsx
<Menubar>
  <MenubarMenu>
    <MenubarTrigger>文件</MenubarTrigger>
    <MenubarContent>
      <MenubarItem>
        新建标签页 <MenubarShortcut>⌘T</MenubarShortcut>
      </MenubarItem>
      <MenubarItem>新建窗口</MenubarItem>
      <MenubarSeparator />
      <MenubarItem>分享</MenubarItem>
      <MenubarSeparator />
      <MenubarItem>打印</MenubarItem>
    </MenubarContent>
  </MenubarMenu>
</Menubar>
```
