## 使用法

```tsx
import {
  Popover,
  PopoverContent,
  PopoverTrigger,
} from "@/components/ui/popover"
```

```tsx
<Popover>
  <PopoverTrigger>開く</PopoverTrigger>
  <PopoverContent>ここにポーバーのコンテンツを配置します。</PopoverContent>
</Popover>
```
