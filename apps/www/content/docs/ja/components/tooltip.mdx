## 使用法

```tsx
import {
  Tooltip,
  TooltipContent,
  TooltipProvider,
  TooltipTrigger,
} from "@/components/ui/tooltip"
```

```tsx
<TooltipProvider>
  <Tooltip>
    <TooltipTrigger>ホバー</TooltipTrigger>
    <TooltipContent>
      <p>ライブラリに追加</p>
    </TooltipContent>
  </Tooltip>
</TooltipProvider>
```
